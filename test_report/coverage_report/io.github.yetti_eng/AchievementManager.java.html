package io.github.yetti_eng;

import java.util.ArrayList;
import java.util.List;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.files.FileHandle;

public class AchievementManager {
    private static final String FILENAME = "achievements.txt";
    private static List<Achievement> achievements;
    
    private static void loadAchievements() {
        FileHandle file = Gdx.files.local(FILENAME);

        if (!file.exists()) throw new RuntimeException(FILENAME + " doesn't exist.");

        achievements = new ArrayList<Achievement>();

        for (String line: file.readString().split("\\r?\\n")) {
            String[] parts = line.split(",");
            if (parts.length != 3) throw new RuntimeException("Invalid achievements file.");
            achievements.add(new Achievement(
                parts[0].trim(),
                parts[1].trim(),
                Boolean.parseBoolean(parts[2].trim())
            ));
        }
    }

    /**
     * Loads achievements from achievements.txt and returns a list of
     * Achievement instances containing the loaded data.
     * @return the list of Achievements
     */
    public static List<Achievement> getAchievements() {
        loadAchievements();
        return achievements;
    }

    private static String getAchievementsString() {
        List<String> achievementStrings = new ArrayList<String>();
        for (Achievement a: achievements) {
            achievementStrings.add(a.toString());
        }
        return String.join("\r\n", achievementStrings);
    }

    private static void saveAchievements() {
        FileHandle file = Gdx.files.local(FILENAME);

        if (!file.exists()) throw new RuntimeException(FILENAME + " doesn't exist.");

        file.writeString(getAchievementsString(), false);
    }

    /**
     * Unlocks the achievement at the given index in achievements.txt
     * (sets third value to true). Updates achievements.txt to reflect this.
     * @param index the line number in achievements.txt of the achievement
     *              to unlock
     */
    public static void unlockAchievement(int index) {
        if (achievements == null) {
            loadAchievements();
        }
        achievements.get(index).unlock();
        saveAchievements();
    }
    
    /**
     * Resets all achievements in achievements.txt (sets the third value on
     * each line to false). Updates achievements.txt to reflect this.
     */
    public static void resetAchievements() {
        if (achievements == null) {
            loadAchievements();
        }
        for (Achievement a : achievements) {
            a.reset();
        }
        saveAchievements();
    }
}
