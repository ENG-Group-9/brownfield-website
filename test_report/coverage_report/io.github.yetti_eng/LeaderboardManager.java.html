package io.github.yetti_eng;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.files.FileHandle;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;

/**
 * NEW for Assessment 2: Manages the reading, writing, and sorting of high scores.
 * This class handles file I/O using LibGDX's local storage to ensure 
 * cross-platform compatibility for the persistent leaderboard system.
 */
public class LeaderboardManager {
    
    /** The name of the file where scores are saved locally. */
    private static final String FILE_NAME = "leaderboard.txt";
    
    /** The maximum number of scores to keep in the leaderboard. */
    private static final int MAX_SCORES = 5;

    /**
     * Represents a single entry in the leaderboard.
     * Encapsulates the player's name and their numeric score.
     */
    public static class ScoreEntry {
        /** The player's name. */
        public String name;
        /** The numeric score achieved. */
        public int score;

        /**
         * Creates a new score entry.
         * @param name The player's name.
         * @param score The final score achieved.
         */
        public ScoreEntry(String name, int score) {
            this.name = name;
            this.score = score;
        }
    }

    /**
     * Retrieves the current top scores from local storage.
     * Creates a dummy file if the leaderboard does not exist.
     * @return An ArrayList of ScoreEntry objects, sorted in descending order.
     */
    public ArrayList<ScoreEntry> fetchScores() {
        FileHandle file = Gdx.files.local(FILE_NAME);
        
        if (!file.exists()) {
            createDummyFile(file);
        }

        ArrayList<ScoreEntry> scores = new ArrayList<>();
        String text = file.readString();
        String[] lines = text.split("\\r?\\n");

        for (String line : lines) {
            String[] parts = line.split(",");
            if (parts.length == 2) {
                try {
                    scores.add(new ScoreEntry(parts[0].trim(), Integer.parseInt(parts[1].trim())));
                } catch (NumberFormatException e) {
                    Gdx.app.error("LeaderboardManager", "Skipping invalid score entry: " + line);
                }                
            }
        }

        sortScores(scores);
        return scores;
    }

    /**
     * Checks if a given score qualifies for the top 5 leaderboard.
     * @param score The score to check.
     * @return true if the score is higher than the lowest on the board, or if board has < 5 entries.
     */
    public boolean isHighScore(int score) {
        ArrayList<ScoreEntry> scores = fetchScores();
        
        if (scores.size() < MAX_SCORES) {
            return true;
        }
        
        return score > scores.get(scores.size() - 1).score;
    }

    /**
     * Adds a new score to the leaderboard and saves the changes to disk.
     * Automatically resorts and trims the list to the top 5 entries.
     * @param name The name of the player.
     * @param score The score achieved.
     */
    public void addScore(String name, int score) {
        ArrayList<ScoreEntry> scores = fetchScores();
        scores.add(new ScoreEntry(name, score));
        
        sortScores(scores);

        if (scores.size() > MAX_SCORES) {
            scores.subList(MAX_SCORES, scores.size()).clear();
        }

        saveScores(scores);
    }

    /**
     * Sorts the list of scores in descending order (highest score first).
     * @param scores The list of scores to sort.
     */
    private void sortScores(ArrayList<ScoreEntry> scores) {
        Collections.sort(scores, new Comparator<ScoreEntry>() {
            @Override
            public int compare(ScoreEntry o1, ScoreEntry o2) {
                return Integer.compare(o2.score, o1.score); 
            }
        });
    }

    /**
     * Resets the leaderboard to the default dummy values.
     */
    public void resetScores() {
        FileHandle file = Gdx.files.local(FILE_NAME);
        createDummyFile(file);
    }

    /**
     * Writes the list of scores to the local file in CSV format.
     * @param scores The list of scores to serialize.
     */
    private void saveScores(ArrayList<ScoreEntry> scores) {
        FileHandle file = Gdx.files.local(FILE_NAME);
        StringBuilder sb = new StringBuilder();
        for (ScoreEntry entry : scores) {
            sb.append(entry.name).append(",").append(entry.score).append("\n");
        }
        file.writeString(sb.toString(), false);
    }

    /**
     * Creates a default leaderboard file with dummy data.
     * @param file The file handle to write to.
     */
    private void createDummyFile(FileHandle file) {
        String dummyData = "Empty,0\nEmpty,0\nEmpty,0\nEmpty,0\nEmpty,0";
        file.writeString(dummyData, false);
    }
}
