package io.github.yetti_eng.entities;

import java.util.ArrayList;

import com.badlogic.gdx.graphics.Texture;

import io.github.yetti_eng.AchievementManager;

public class Player extends Entity {
    public static final float DEFAULT_SPEED = 6f;

    private final ArrayList<Item> inventory;
    private Boolean touchedWall;

    public Player(Texture tex, float x, float y) {
        super(tex, x, y, 0.9f, 1.6f, DEFAULT_SPEED, false);
        this.inventory = new ArrayList<>();
        touchedWall = false;
    }

    public Player(Texture tex, float x, float y, ArrayList<Item> initialInventory) {
        super(tex, x, y, 0.9f, 1.6f, DEFAULT_SPEED, false);
        this.inventory = initialInventory;
    }

    public boolean hasUsedItem(String itemID) {
        return inventory.stream().anyMatch(i -> i.ID.equals(itemID));
    }

    // MODIFIED (Assessment 2)
    // To allow for achievements related to items collected
    public void addItemToInventory(Item item) {
        inventory.add(item);

        if (countItemsWithID("long_boi") == 3) {
            AchievementManager.unlockAchievement(0);
        }

        if (countItemsWithID("coffee") == 5) {
            AchievementManager.unlockAchievement(1);
        }
    }

    public long countItemsWithID(String id) {
        return inventory.stream().filter(i -> i.getID() == id).count();
    }

    public void touchWall() {
        touchedWall = true;
    }

    public Boolean getTouchedWall() {
        return touchedWall;
    }
}
