package io.github.yetti_eng.entities;

import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.math.MathUtils;
import com.badlogic.gdx.math.Vector2;

import io.github.yetti_eng.EventCounter;
import io.github.yetti_eng.MapManager;
import io.github.yetti_eng.screens.GameScreen;

/**
 * Represents a "Social Sec" NPC that wanders the bottom left of the map.
 * Interacting with them results in a time penalty for the player.
 */
public class SocialSec extends Entity {

    /** Timer tracks when the NPC should next change direction. */
    private float changeDirectionTimer = 0;
    /** Tracks if the NPC has already triggered their time-wasting event. */
    private boolean hasWastedTime = false;

    /**
     * Initialises the Social Sec entity.
     * @param tex The texture for the NPC sprite.
     * @param x The starting X-coordinate.
     * @param y The starting Y-coordinate.
     */
    public SocialSec(Texture tex, float x, float y) {
        super(tex, x, y, 1.5f, 1.5f, 3.5f, false); 
        pickRandomDirection();
    }

    /**
     * Updates the NPC's wandering behaviour.
     * @param delta The time in seconds since the last update.
     * @param mapManager Used to check for collisions with the environment.
     */
    public void wander(float delta, MapManager mapManager) {
        changeDirectionTimer -= delta;

        if (changeDirectionTimer <= 0) {
            pickRandomDirection();
        }

        float oldX = getX();
        float oldY = getY();

        doMove(delta);

        if (mapManager.isRectInvalid(getHitbox())) {
            setPosition(oldX, oldY);
            pickRandomDirection();
        }
    }

    /**
     * Selects a new random movement direction and sets a random duration.
     */
    private void pickRandomDirection() {
        float angle = MathUtils.random(0f, 360f);
        Vector2 dir = new Vector2(1, 0).setAngleDeg(angle);
        
        resetMovement();
        addMovement(dir.x, dir.y);
        changeDirectionTimer = MathUtils.random(1.0f, 3.0f);
    }

    /**
     * Checks if the NPC interacts with the player.
     * If the interaction hasn't occurred yet, reduces the game timer.
     * @param screen The current game screen.
     * @return true if a time-wasting interaction was triggered.
     */
    public boolean checkInteraction(GameScreen screen) {
        if (!hasWastedTime) {
            hasWastedTime = true;
            
            EventCounter.incrementNegative();
            
            screen.spawnInteractionMessage("A social sec wasted your time discussing a social");
            
            screen.getGame().timer.reduceTime(30); 
            
            return true;
        }
        return false;
    }
}
