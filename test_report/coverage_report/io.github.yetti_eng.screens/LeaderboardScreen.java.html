package io.github.yetti_eng.screens;

import java.util.ArrayList;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.LeaderboardManager;
import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

/**
 * NEW (Assessment 2): Displays the High Score Leaderboard to the user.
 *
 * This screen is a new addition required by the product brief to allow players to view
 * the top 5 scores achieved in the game. It gets data from LeaderboardManager
 * and renders it using the game's UI styling.
 *
 * Requirements:
 * - UR_MAIN_MENU (Navigation to Leaderboard)
 * - FR_LEADERBOARD_SCREEN (Displaying the top 5 names and scores)
 */
public class LeaderboardScreen implements Screen {
    private final YettiGame game;
    private final Stage stage;
    private final LeaderboardManager leaderboardManager;

    /**
     * Initialises the Leaderboard screen.
     *
     * @param game The main game instance.
     */
    public LeaderboardScreen(final YettiGame game) {
        this.game = game;
        this.stage = new Stage(game.viewport, game.batch);
        this.leaderboardManager = new LeaderboardManager();
    }

    /**
     * Called when this screen becomes the current screen.
     *
     * Sets up the UI stage:
     * 1. Creates and positions the Title Label.
     * 2. Fetches the sorted list of high scores from LeaderboardManager.
     * 3. Iterates through the scores to create dynamic Labels for the Top 5.
     * 4. Adds a "Back" button to return to the Main Menu.
     */
    @Override
    public void show() {
        Gdx.input.setInputProcessor(stage);

        Label titleLabel = new Label("Top 5 Scores", new Label.LabelStyle(game.font, Color.WHITE));
        titleLabel.setPosition(scaled(16) / 2, scaled(7.5f), Align.center);
        stage.addActor(titleLabel);

        ArrayList<LeaderboardManager.ScoreEntry> scores = leaderboardManager.fetchScores();
        float startY = 6.0f;

        for (int i = 0; i < scores.size(); i++) {
            LeaderboardManager.ScoreEntry entry = scores.get(i);
            String text =String.format("%d. %s - %d", i + 1, entry.name, entry.score);
            
            Label scoreLabel = new Label(text, new Label.LabelStyle(game.fontBorderedSmall, Color.WHITE));
            scoreLabel.setPosition(scaled(16) / 2, scaled(startY - (i * 0.8f)), Align.center);
            stage.addActor(scoreLabel);
        }

        // Button to get back to the main menu
        TextButton.TextButtonStyle style = new TextButton.TextButtonStyle(null, null, null, game.font);
        TextButton backButton = new TextButton("Back", style);
        backButton.setPosition(scaled(16) / 2, scaled(1.0f), Align.center);
        backButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                game.setScreen(new MenuScreen(game));
                dispose();
                return true;
            }
        });
        stage.addActor(backButton);
    }

    /**
     * Renders the background and the UI stage.
     *
     * @param delta The time in seconds since the last render.
     */
    @Override
    public void render(float delta) {
        ScreenUtils.clear(0.15f, 0.15f, 0.2f, 1f);
        game.viewport.apply();
        stage.draw();
    }

    /**
     * Updates the UI viewport dimensions.
     * @param width The new width in pixels.
     * @param height The new height in pixels.
     */
    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    /**
     * Releases all resources used by the screen.
     */
    @Override
    public void dispose() {
        stage.dispose();
    }
}
