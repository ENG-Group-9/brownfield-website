package io.github.yetti_eng.screens;

import java.util.List;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.ScrollPane;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.Achievement;
import io.github.yetti_eng.AchievementManager;
import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

/**
 * Screen that displays the player's achievements and their unlocked status.
 * Features a scrollable list of achievements.
 */
public class AchievementsScreen implements Screen {

    /** The main game instance. */
    private final YettiGame game;
    /** The stage containing UI actors. */
    private Stage stage;
    /** Scrollable pane for the achievement list. */
    private ScrollPane scrollPane;

    /**
     * Initialises the achievements screen.
     * @param game The main game instance.
     */
    public AchievementsScreen(YettiGame game) {
        this.game = game;
    }

    /**
     * Called when this screen becomes the current screen.
     * Sets up the layout, title, achievement list, and back button.
     */
    @Override
    public void show() {
        stage = new Stage(game.viewport, game.batch);
        Gdx.input.setInputProcessor(stage);

        Table mainTable = new Table();
        mainTable.setFillParent(true);

        Label title = new Label("Achievements", new Label.LabelStyle(game.font, Color.WHITE));
        mainTable.add(title).pad(scaled(0.5f)).top().row();

        Table listTable = new Table();
        listTable.top();

        loadAchievements(listTable);

        ScrollPane.ScrollPaneStyle scrollStyle = new ScrollPane.ScrollPaneStyle();
        scrollPane = new ScrollPane(listTable, scrollStyle);
        scrollPane.setScrollingDisabled(true, false);
        
        mainTable.add(scrollPane).expand().fill().padBottom(scaled(2.0f)).row();

        stage.addActor(mainTable);

        TextButton.TextButtonStyle style = new TextButton.TextButtonStyle(null, null, null, game.font);
        TextButton backButton = new TextButton("Back", style);
        
        backButton.setPosition(scaled(16) / 2, scaled(1.0f), Align.center);
        
        backButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                game.setScreen(new MenuScreen(game));
                dispose();
                return true;
            }
        });
        stage.addActor(backButton);
    }

    /**
     * Fetches achievements from the manager and populates the UI table.
     * Includes the "Graduate" achievement if all others are unlocked.
     * @param table The table to add achievement labels to.
     */
    private void loadAchievements(Table table) {
        List<Achievement> achievements = AchievementManager.getAchievements();
        
        boolean allUnlocked = true;

        for (Achievement a: achievements) {
            if (!a.getUnlocked()) {
                allUnlocked = false;
            }
            addAchievement(table, a.getName(), a.getDescription(), a.getUnlocked());
        }

        addAchievement(table, "Graduate", "Obtain all other achievements.", allUnlocked);
    }

    /**
     * Adds a single achievement row to the list table.
     * @param table The UI table to add to.
     * @param name The name of the achievement.
     * @param description The description of how to unlock it.
     * @param unlocked Whether the achievement is currently unlocked.
     */
    private void addAchievement(Table table, String name, String description, boolean unlocked) {
        Color color = unlocked ? Color.GREEN : Color.RED;
        
        Label.LabelStyle nameStyle = new Label.LabelStyle(game.fontMedium, color);
        Label.LabelStyle descStyle = new Label.LabelStyle(game.fontSmall, Color.WHITE);

        Label nameLabel = new Label(name, nameStyle);
        Label descLabel = new Label(description, descStyle);
        descLabel.setWrap(true);
        descLabel.setAlignment(Align.center);

        table.add(nameLabel).padTop(scaled(0.2f)).row();
        table.add(descLabel).width(game.viewport.getWorldWidth() * 0.8f).padBottom(scaled(0.2f)).row(); 
    }

    /**
     * Renders the achievements screen.
     * Handles keyboard scrolling input for the achievement list.
     * @param delta The time in seconds since the last render.
     */
    @Override
    public void render(float delta) {
        ScreenUtils.clear(0f, 0f, 0f, 1f);

        float scrollSpeed = scaled(5f); 
        
        if (Gdx.input.isKeyPressed(Input.Keys.UP)) {
            scrollPane.setScrollY(scrollPane.getScrollY() - scrollSpeed * delta);
        }
        if (Gdx.input.isKeyPressed(Input.Keys.DOWN)) {
            scrollPane.setScrollY(scrollPane.getScrollY() + scrollSpeed * delta);
        }

        stage.act(delta);
        stage.draw();
    }

    /**
     * Updates the UI viewport dimensions.
     * @param width The new width in pixels.
     * @param height The new height in pixels.
     */
    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    /**
     * Releases resources used by the stage.
     */
    @Override
    public void dispose() {
        stage.dispose();
    }
}
