package io.github.yetti_eng.screens;

import java.util.ArrayList;
import java.util.List;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Image;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.TextureRegionDrawable;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.Scaling;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

/**
 * A screen that displays a "How to Play" tutorial to the user.
 *
 * The tutorial contains information such as the different events in the game and the controls
 */
public class HowToPlayScreen implements Screen {
    private final YettiGame game;
    private final Stage stage;
    private final List<Texture> slides;
    private int currentSlideIndex;
    private Image slideImage;
    private Label pageCounterLabel;

    /**
     * Constructs a new HowToPlayScreen.
     *
     * @param game The main game object, used for screen transitions and accessing shared resources.
     */
    public HowToPlayScreen(final YettiGame game) {
        this.game = game;
        this.stage = new Stage(game.viewport, game.batch);
        this.slides = new ArrayList<>();
        this.currentSlideIndex = 0;
    }

    /**
     * Called when this screen becomes the current screen for the game.
     *
     * This method initialises the stage, loads the tutorial images from the assets,
     * and sets up the UI layout including the slide image, navigation buttons, and page counter.
     * 
     * Also has a fallback for if a tutorial page is missing
     */
    @Override
    public void show() {
        Gdx.input.setInputProcessor(stage);

        for (int i = 1; i <= 4; i++) {
            try {
                slides.add(new Texture(Gdx.files.internal("tutorial/tutorial_" + i + ".png")));
            } catch (Exception e) {
                Gdx.app.log("HowToPlayScreen", "Missing tutorial_" + i + ".png, using fallback.");
                try {
                    slides.add(new Texture(Gdx.files.internal("character/yeti.png")));
                } catch (Exception ignored) {}
            }
        }

        Table rootTable = new Table();
        rootTable.setFillParent(true);
        stage.addActor(rootTable);

        if (!slides.isEmpty()) {
            slideImage = new Image(slides.get(0));
            slideImage.setScaling(Scaling.fit);
            rootTable.add(slideImage).grow().row();
        } else {
            rootTable.add().grow().row();
        }

        Table controlsTable = new Table();

        TextButton.TextButtonStyle style = new TextButton.TextButtonStyle(null, null, null, game.fontMedium);

        TextButton exitButton = new TextButton("Exit", style);
        exitButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                game.setScreen(new MenuScreen(game));
                dispose();
                return true;
            }
        });

        Label.LabelStyle counterStyle = new Label.LabelStyle(game.fontMedium, Color.WHITE);
        pageCounterLabel = new Label("1/" + slides.size(), counterStyle);
        pageCounterLabel.setAlignment(Align.center);

        TextButton nextButton = new TextButton("Next", style);
        nextButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                nextSlide();
                return true;
            }
        });

        controlsTable.add(exitButton).left().pad(scaled(0.5f));
        controlsTable.add(pageCounterLabel).expandX().center();
        controlsTable.add(nextButton).right().pad(scaled(0.5f));

        rootTable.add(controlsTable).growX().padBottom(scaled(0.2f));
    }

    /**
     * Advances the tutorial to the next slide.
     *
     * If the current slide is the last one, calling this method will return the user
     * to the main menu. Otherwise, it updates the displayed image and the page counter.
     */
    private void nextSlide() {
        currentSlideIndex++;
        if (currentSlideIndex >= slides.size()) {
            game.setScreen(new MenuScreen(game));
            dispose();
        } else {
            slideImage.setDrawable(new TextureRegionDrawable(slides.get(currentSlideIndex)));
            updatePageCounter();
        }
    }

    /**
     * Updates the text of the page counter label to reflect the current slide index.
     */
    private void updatePageCounter() {
        if (pageCounterLabel != null) {
            pageCounterLabel.setText((currentSlideIndex + 1) + "/" + slides.size());
        }
    }

    /**
     * Renders the screen.
     *
     * Clears the screen with a black background and draws the stage containing the UI elements.
     *
     * @param delta The time in seconds since the last render.
     */
    @Override
    public void render(float delta) {
    ScreenUtils.clear(0.15f, 0.15f, 0.2f, 1f);
        game.viewport.apply();
        stage.draw();
    }

    /**
     * Called when the application is resized.
     *
     * Updates the game viewport to match the new screen dimensions.
     *
     * @param width  The new width in pixels.
     * @param height The new height in pixels.
     */
    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    @Override
    public void dispose() {
        stage.dispose();
        for (Texture t : slides) {
            t.dispose();
        }
    }
}
