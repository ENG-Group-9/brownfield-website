package io.github.yetti_eng.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.scenes.scene2d.Actor;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.Slider;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.ChangeListener;
import com.badlogic.gdx.scenes.scene2d.utils.TextureRegionDrawable;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.AchievementManager;
import io.github.yetti_eng.LeaderboardManager;
import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

public class SettingsScreen implements Screen {
    private final YettiGame game;
    private final Stage stage;

    private Slider volumeSlider;

    private Texture sliderTexture;
    private Texture knobTexture;

    public SettingsScreen(final YettiGame game) {
        this.game = game;
        stage = new Stage(game.viewport, game.batch);
    }

    @Override
    public void show() {
        Gdx.input.setInputProcessor(stage);

        Table table = new Table();
        table.setFillParent(true);
        stage.addActor(table);

        sliderTexture = new Texture("ui/slider_background.png");
        knobTexture = new Texture("ui/slider_knob.png");

        Label.LabelStyle titleStyle = new Label.LabelStyle(game.font, Color.WHITE);
        Label titleLabel = new Label("Settings", titleStyle);
        titleLabel.setAlignment(Align.center);

        Label volumeLabel = new Label("Volume", new Label.LabelStyle(game.fontBorderedSmall, Color.WHITE));
        
        volumeSlider = new Slider(0.0f, 1.0f, 0.01f, false,
            new Slider.SliderStyle(new TextureRegionDrawable(sliderTexture), new TextureRegionDrawable(knobTexture))
        );
        volumeSlider.setValue(game.volume);
        volumeSlider.addListener(new ChangeListener() {
            @Override
            public void changed(ChangeEvent event, Actor actor) {
                game.volume = volumeSlider.getValue();
            }
        });

        TextButton.TextButtonStyle buttonStyle = new TextButton.TextButtonStyle(null, null, null, game.font);

        TextButton resetButton = new TextButton("Reset Leaderboard", buttonStyle);
        resetButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                LeaderboardManager manager = new LeaderboardManager();
                manager.resetScores();
                Gdx.app.log("Settings", "Leaderboard reset to defaults.");
                return true;
            }
        });

        TextButton menuButton = new TextButton("Return to Menu", buttonStyle);
        menuButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                game.setScreen(new MenuScreen(game));
                dispose();
                return true;
            }
        });

        TextButton resetAchievementsButton = new TextButton("Reset Achievements", buttonStyle);
        resetAchievementsButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                AchievementManager.resetAchievements();
                Gdx.app.log("Settings", "Achievements reset to defaults.");
                return true;
            }
        });

        float gap = scaled(0.5f);
        
        table.add(titleLabel).padBottom(gap).row();
        table.add(volumeLabel).padBottom(scaled(0.1f)).row();
        table.add(volumeSlider).width(scaled(8)).padBottom(gap).row(); 
        table.add(resetButton).width(scaled(6)).padBottom(gap).row();
        table.add(resetAchievementsButton).width(scaled(6)).padBottom(gap).row();
        table.add(menuButton).padTop(scaled(0.5f));
    }

    /**
     * Renders the settings screen elements.
     *
     * Clears the background and draws the Stage.
     *
     * @param delta The time in seconds since the last render.
     */
    @Override
    public void render(float delta) {
        ScreenUtils.clear(0.4f, 0.4f, 0.4f, 1f);
        game.viewport.apply();    
        stage.draw();
    }

    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    @Override
    public void dispose() {
        stage.dispose();
        sliderTexture.dispose();
        knobTexture.dispose();
    }
}
