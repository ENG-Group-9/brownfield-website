package io.github.yetti_eng.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

public class WinScreen implements Screen {
    private final YettiGame game;
    private final Stage stage;
    
    // MODIFIED (Assessment 2): Added Score Field 
    // Stores the final score calculated to display to the winner.
    private final int score;

    // MODIFIED (Assessment 2)
    // Now requires the score parameter. This ensures that the scoring logic 
    // from UR_SCORING is visible to the player when completing the level.
    public WinScreen(final YettiGame game, int score) {
        this.game = game;
        this.score = score;
        this.stage = new Stage(game.viewport, game.batch);
    }

    public WinScreen(final YettiGame game) {
        this(game, 0);
    }

    @Override
    public void show() {
        Gdx.input.setInputProcessor(stage);

        // Title
        Label titleLabel = new Label("YOU ESCAPED!", new Label.LabelStyle(game.font, Color.GREEN));
        titleLabel.setPosition(scaled(16) / 2, scaled(7f), Align.center);
        stage.addActor(titleLabel);

        // NEW (Assessment 2) Score Display
        // Creates a UI label to show the final score.
        // Satisfies UR_ENDING.
        Label scoreLabel = new Label("Final Score: " + score, new Label.LabelStyle(game.fontBordered, Color.WHITE));
        scoreLabel.setPosition(scaled(16) / 2, scaled(5.5f), Align.center);
        stage.addActor(scoreLabel);

        // Menu Button
        TextButton.TextButtonStyle style = new TextButton.TextButtonStyle(null, null, null, game.font);
        TextButton menuButton = new TextButton("Main Menu", style);
        menuButton.setPosition(scaled(16) / 2, scaled(2f), Align.center);
        menuButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                game.setScreen(new MenuScreen(game));
                dispose();
                return true;
            }
        });
        stage.addActor(menuButton);
    }

    @Override
    public void render(float delta) {
        ScreenUtils.clear(0.15f, 0.15f, 0.2f, 1f);
        game.viewport.apply();
        stage.draw();
    }

    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    @Override
    public void dispose() {}
}
