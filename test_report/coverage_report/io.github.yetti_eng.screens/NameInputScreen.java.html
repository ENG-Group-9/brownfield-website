package io.github.yetti_eng.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.InputListener;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.ui.TextField;
import com.badlogic.gdx.scenes.scene2d.utils.TextureRegionDrawable;
import com.badlogic.gdx.utils.Align;
import com.badlogic.gdx.utils.ScreenUtils;

import io.github.yetti_eng.LeaderboardManager;
import io.github.yetti_eng.YettiGame;
import static io.github.yetti_eng.YettiGame.scaled;

/**
 * NEW (Assessment 2): Screen for entering player details after a high score.
 *
 * This screen is triggered automatically when a player's
 * final score qualifies for the Top 5 Leaderboard. It provides a text field for
 * the user to input their name, which is then saved.
 *
 * Associated Requirements:
 * UR_MAIN_MENU
 * UR_SCORING
 */
public class NameInputScreen implements Screen {
    
    private final YettiGame game;
    private final Stage stage;
    private final int finalScore;
    private TextField nameField;
    private Label errorLabel;
    private com.badlogic.gdx.graphics.Texture cursorTexture;
    private com.badlogic.gdx.graphics.Texture fieldBackgroundTexture;

    /**
     * Initialises the Name Input screen.
     *
     * @param game The main game instance.
     * @param score The final score achieved by the player, to be saved.
     */
    public NameInputScreen(final YettiGame game, int score) {
        this.game = game;
        this.finalScore = score;
        this.stage = new Stage(game.viewport, game.batch);
    }

    /**
     * Called when this screen becomes the current screen.
     * Sets up UI elements including the header, score display, and input field.
     */
    @Override
    public void show() {
        Gdx.input.setInputProcessor(stage);

        com.badlogic.gdx.graphics.Pixmap pixmap = new com.badlogic.gdx.graphics.Pixmap(1, 1, com.badlogic.gdx.graphics.Pixmap.Format.RGBA8888);
        pixmap.setColor(Color.WHITE);
        pixmap.fill();
        cursorTexture = new com.badlogic.gdx.graphics.Texture(pixmap);
        
        pixmap.setColor(new Color(0.2f, 0.2f, 0.2f, 0.5f)); 
        pixmap.fill();
        fieldBackgroundTexture = new com.badlogic.gdx.graphics.Texture(pixmap);
        pixmap.dispose();

        Label titleLabel = new Label("NEW TOP 5 SCORE!", new Label.LabelStyle(game.font, Color.YELLOW));
        titleLabel.setPosition(scaled(16) / 2, scaled(7f), Align.center);
        stage.addActor(titleLabel);

        Label scoreLabel = new Label("Score: " + finalScore, new Label.LabelStyle(game.fontBordered, Color.WHITE));
        scoreLabel.setPosition(scaled(16) / 2, scaled(6f), Align.center);
        stage.addActor(scoreLabel);

        Label promptLabel = new Label("Enter your name:", new Label.LabelStyle(game.fontBorderedSmall, Color.WHITE));
        promptLabel.setPosition(scaled(16) / 2, scaled(5f), Align.center);
        stage.addActor(promptLabel);

        TextField.TextFieldStyle fieldStyle = new TextField.TextFieldStyle();
        fieldStyle.font = game.fontBorderedSmall;
        fieldStyle.fontColor = Color.WHITE;
        
        fieldStyle.cursor = new TextureRegionDrawable(new com.badlogic.gdx.graphics.g2d.TextureRegion(cursorTexture));
        fieldStyle.cursor.setMinWidth(2); 
        fieldStyle.background = new TextureRegionDrawable(new com.badlogic.gdx.graphics.g2d.TextureRegion(fieldBackgroundTexture));

        nameField = new TextField("", fieldStyle);
        nameField.setPosition(scaled(16) / 2 - 150, scaled(4f)); 
        nameField.setSize(300, 50);
        nameField.setAlignment(Align.center);
        
        com.badlogic.gdx.utils.Timer.schedule(new com.badlogic.gdx.utils.Timer.Task() {
            @Override
            public void run() {
                stage.setKeyboardFocus(nameField);
            }
        }, 0.5f); // 0.5 second delay so that if the user is using WASD to move this doesn't bleed into the name input
        
        stage.addActor(nameField);

        errorLabel = new Label("", new Label.LabelStyle(game.fontBorderedSmall, Color.RED));
        errorLabel.setWidth(scaled(16)); 
        errorLabel.setAlignment(Align.center); 
        errorLabel.setPosition(0, scaled(3f)); 
        stage.addActor(errorLabel);

        TextButton.TextButtonStyle style = new TextButton.TextButtonStyle(null, null, null, game.font);
        TextButton submitButton = new TextButton("Submit", style);
        submitButton.setPosition(scaled(16) / 2, scaled(2f), Align.center);
        submitButton.addListener(new InputListener() {
            @Override
            public boolean touchDown(InputEvent event, float x, float y, int pointer, int button) {
                submitScore();
                return true;
            }
        });
        stage.addActor(submitButton);
    }

    /**
     * Captures the input name, validates it, saves the score, and transitions to the Leaderboard.
     * Validates that the name is not empty, within length limits, and alphanumeric.
     */
    private void submitScore() {
        String name = nameField.getText().trim();
        
        if (name.isEmpty()) {
            errorLabel.setText("Name cannot be empty!");
            return;
        }

        if (name.length() > 8) {
            errorLabel.setText("Name too long! (Max 8)");
            return;
        }

        if (!name.matches("^[a-zA-Z0-9]*$")) {
            errorLabel.setText("Alphanumeric only! (No symbols)");
            return;
        }
        
        errorLabel.setText("");
        
        LeaderboardManager manager = new LeaderboardManager();
        manager.addScore(name, finalScore);

        game.setScreen(new LeaderboardScreen(game));
        dispose();
    }

    /**
     * Renders the screen background and the UI stage.
     * @param delta The time in seconds since the last render.
     */
    @Override
    public void render(float delta) {
        ScreenUtils.clear(0.15f, 0.15f, 0.2f, 1f);
        game.viewport.apply();
        stage.draw();
    }

    /**
     * Updates the UI viewport dimensions.
     * @param width The new width in pixels.
     * @param height The new height in pixels.
     */
    @Override
    public void resize(int width, int height) {
        game.viewport.update(width, height, true);
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {}

    /**
     * Releases all resources used by the screen.
     */
    @Override
    public void dispose() {
        stage.dispose();
        if (cursorTexture != null) cursorTexture.dispose();
        if (fieldBackgroundTexture != null) fieldBackgroundTexture.dispose();
    }
}
