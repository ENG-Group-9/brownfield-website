package io.github.yetti_eng.events;

import com.badlogic.gdx.utils.Timer;

import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;

public class WallEvent extends Event {

    private boolean canTrigger = true;

    //CHANGED: Added method overload for timer object (unit testing, AS2)

    @Override
    public boolean activate(GameScreen screen, Player player, Item item) {
        return this.activate(screen, player, item, new Timer());
    }

    public boolean activate(GameScreen screen, Player player, Item item, Timer timer) {
        if (!canTrigger) {
            return false;
        }

        player.touchWall();
        
        canTrigger = false;
        screen.getStubSfx().play(screen.getGame().volume);
        screen.spawnInteractionMessage("Ouch! You stubbed your toe (" + getScoreModifier() + ")");

        timer.scheduleTask(new Timer.Task() {
            @Override
            public void run() {
                canTrigger = true;
            }
        }, 2f);

        return false;
    }

    @Override
    public int getScoreModifier() {
        return -10;
    }

    //CHANGED: Added getters/setters for canTrigger variable (unit testing)
    public void setTrigger(boolean trigger) {
        this.canTrigger = trigger;
    }

    public boolean getTrigger() {
        return this.canTrigger;
    }
}
