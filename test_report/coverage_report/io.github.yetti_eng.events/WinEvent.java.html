package io.github.yetti_eng.events;

import io.github.yetti_eng.EventCounter;
import io.github.yetti_eng.LeaderboardManager;
import io.github.yetti_eng.YettiGame;
import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;
import io.github.yetti_eng.screens.NameInputScreen;
import io.github.yetti_eng.screens.WinScreen;

public class WinEvent extends Event {

    private final LeaderboardManager leaderboardManager;

    // NEW (AS2): Add constructor to pass objects for unit testing
    public WinEvent(LeaderboardManager leaderboardManager) {
        this.leaderboardManager = leaderboardManager;
    }

    @Override
    public boolean activate(final GameScreen screen, Player player, Item item) {

        YettiGame game = screen.getGame();
        
        // NEW: Calculate Final Score (Assessment 2 Requirement: UR_SCORING)
        int timeRemaining = game.timer.getRemainingTime();
        int positive = EventCounter.getPositiveCount();
        int negative = EventCounter.getNegativeCount();
        
        int score = (timeRemaining * 10) + (positive * 100) - (negative * 50);
        if (score < 0) score = 0;

        // NEW (AS2): Leaderboard Integration        
        if (leaderboardManager.isHighScore(score)) {
            game.setScreen(this.createNameInputScreen(game, score));
        } else {
            game.setScreen(this.createWinScreen(game, score));
        }
        
        return true;
    }

    // NEW (AS2): Added functions for screen setting to support unit testing
    public NameInputScreen createNameInputScreen(YettiGame game, int score) {
        return new NameInputScreen(game, score);
    }

    public WinScreen createWinScreen(YettiGame game, int score) {
        return new WinScreen(game, score);
    }

    @Override
    public int getScoreModifier() {
        return 0; 
    }
}
