package io.github.yetti_eng.events;

import io.github.yetti_eng.entities.Entity;
import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;

/**
 * Event that triggers when a player realises they have forgotten their backpack.
 * Enables the hidden backpack item on the map for collection.
 */
public class BackpackTriggerEvent extends Event {

    /**
     * Activates the backpack trigger event.
     * Disables the trigger zone and reveals the hidden backpack item.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The invisible trigger item.
     * @return true if the event was successfully activated.
     */
    @Override
    public boolean activate(GameScreen screen, Player player, Item item) {
        if (!item.isUsed()) {
            item.disable();
            
            for (Entity e : screen.getEntities()) {
                if (e instanceof Item && "backpack".equals(((Item)e).getID())) {
                    e.show();
                    e.enable();
                    break;
                }
            }
            
            screen.spawnInteractionMessage("You've forgotten your backpack!");
            
            return true;
        }
        return false;
    }

    /**
     * Returns the score modifier for triggering the backpack realisation.
     * @return Zero, as the realisation itself grants no points.
     */
    @Override
    public int getScoreModifier() {
        return 0; 
    }
}
