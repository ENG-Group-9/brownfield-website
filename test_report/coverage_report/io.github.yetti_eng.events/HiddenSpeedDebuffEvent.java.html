package io.github.yetti_eng.events;

import com.badlogic.gdx.utils.Timer;

import io.github.yetti_eng.EventCounter;
import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;

/**
 * A hidden event representing a loose floorboard that the player trips on.
 * Reduces the player's speed temporarily and deducts points.
 */
public class HiddenSpeedDebuffEvent extends Event {

    /**
     * Activates the hidden speed debuff event using a new Timer.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The floorboard item triggered.
     * @return true if the event was successfully activated.
     */
    @Override
    public boolean activate(final GameScreen screen, final Player player, Item item) {
        return this.activate(screen, player, item, new Timer());
    }

    /**
     * Activates the hidden speed debuff event.
     * Triggers sound effects, reveals the floorboard, and sets a temporary speed penalty.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The floorboard item triggered.
     * @param timer The timer used to schedule the speed recovery.
     * @return true if the event was successfully activated.
     */
    public boolean activate(final GameScreen screen, final Player player, Item item, Timer timer) {
        // Only trigger if the item hasn't been used yet
        if (!item.isUsed()) {
            EventCounter.incrementHidden();
            EventCounter.incrementNegative();
            
            // Reveal the item (the loose floorboard) so the player knows what happened
            item.show(); 
            
            // Play a sound effect
            screen.getFloorboardSfx().play(screen.getGame().volume);            
            screen.spawnInteractionMessage("Stumbled on a loose floorboard! (-Speed)");

            // Apply Slow Effect: Set speed to 2.0
            player.setSpeed(2f);

            // Schedule the reset of the speed after 3 seconds
            timer.scheduleTask(new Timer.Task() {
                @Override
                public void run() {
                    player.setSpeed(Player.DEFAULT_SPEED);
                    screen.spawnInteractionMessage("You recovered your balance.");
                }
            }, 3f);

            return true;
        }
        return false; // Item already used
    }

    /**
     * Returns the score modifier for "tripping".
     * @return A negative integer representing the penalty.
     */
    @Override
    public int getScoreModifier() {
        return -50; 
    }
}
