package io.github.yetti_eng.events;

import com.badlogic.gdx.utils.Timer;

import io.github.yetti_eng.EventCounter;
import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;

/**
 * Event representing a caffeine hit that boosts player speed.
 */
public class SpeedBoostEvent extends Event {

    /**
     * Activates the speed boost event using a new Timer.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The coffee item collected.
     * @return true if the event was successfully activated.
     */
    @Override
    public boolean activate(final GameScreen screen, final Player player, Item item) {
        return this.activate(screen, player, item, new Timer());
    }

    /**
     * Activates the speed boost event.
     * Disables the item, plays a sound, and sets the player speed to 8.0 for 5 seconds.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The coffee item collected.
     * @param timer The timer used to schedule the speed reset.
     * @return true if the event was successfully activated.
     */
    public boolean activate(final GameScreen screen, final Player player, Item item, Timer timer) {
        if (!item.isUsed()) {
            EventCounter.incrementPositive();

            item.hide();
            item.disable();

            screen.getSipSfx().play(screen.getGame().volume);

            screen.spawnInteractionMessage("Caffeine Hit! (+Speed)");

            player.setSpeed(8f);

            timer.scheduleTask(new Timer.Task() {
                @Override
                public void run() {
                    player.setSpeed(Player.DEFAULT_SPEED);
                    screen.spawnInteractionMessage("The caffeine wore off.");
                }
            }, 5f);

            return true;
        }
        return false;
    }

    /**
     * Returns the score modifier for getting a speed boost.
     * @return A positive integer representing the points gained.
     */
    @Override
    public int getScoreModifier() {
        return 100;
    }
}
