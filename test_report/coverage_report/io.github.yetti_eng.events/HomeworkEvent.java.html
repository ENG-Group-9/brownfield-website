package io.github.yetti_eng.events;

import io.github.yetti_eng.EventCounter;
import io.github.yetti_eng.entities.Entity;
import io.github.yetti_eng.entities.Item;
import io.github.yetti_eng.entities.Player;
import io.github.yetti_eng.screens.GameScreen;

/**
 * Event representing the collection of homework.
 * Triggers score increases and potentially alters the Dean's arrival time.
 */
public class HomeworkEvent extends Event {

    /**
     * Activates the homework collection event.
     * Disables the item, plays a sound, and checks if collection was on time.
     * @param screen The current game screen.
     * @param player The player instance.
     * @param item The homework item being collected.
     * @return true if the event was successfully activated.
     */
    @Override
    public boolean activate(GameScreen screen, Player player, Item item) {
        if (!item.isUsed()) {
            item.disable();

            for (Entity e : screen.getEntities()) {
                if (e instanceof Item && "homework_prop".equals(((Item)e).getID())) {
                    e.hide();
                    e.disable();
                }
            }
            
            screen.getPaperSfx().play(screen.getGame().volume);
            
            if (screen.getGame().timer.getRemainingTime() > 180) {
                screen.spawnInteractionMessage("Homework collected on time");
                screen.setDeanTriggerTime(60); 
                EventCounter.incrementPositive();
            } else {
                 screen.spawnInteractionMessage("Homework collected late!");
            }

            return true;
        }
        return false;
    }

    /**
     * Returns the score modifier for collecting homework.
     * @return A positive integer representing the points gained.
     */
    @Override
    public int getScoreModifier() {
        return 500;
    }
}
